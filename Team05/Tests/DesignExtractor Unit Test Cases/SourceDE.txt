procedure First{ 
	x = 2;			
	z = 3;			
	call Second;
	y = x;
	call Fifth;}	

procedure Second{ 
	while i {		
		x = x + 2 * y;	
		call Third;		
		i = i - 1;}		
	call Fourth;
	if x then{		
		x = x + 1;}		
	else{			
		z = 1;}		
	z = z + x + i;		
	y = z + 2;		
	x = x * y + z;}	

procedure Third{ 
	z = 5;			
	v = z;}		

procedure Fourth{ 
	x = 5;			
	z = v;}		

procedure Fifth{

	z = 5;
